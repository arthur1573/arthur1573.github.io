// welcome to vim text editor
vim




// looking for help
vim
F1
:help




// open command mode
:



// quit help vim help page
:q



// command mode: force quit
:q!



// command mode: save write and quit
:wq





// opening vim will open default: view mode
// view mode: go up, down, right, left
jkhl






// open insert mode(write mode)
i





// view mode: undo
u





// view mode: redo
Ctrl + R
^R







// view mode: delete single line
dd





// command mode: delete all line
:%d




// view mode: copy single line
yy




// view mode: paste
p



// open visual mode, move to select
v


// visual mode: copy
y


// visual mode: delete
d








// command mode: line number
:set number



// command mode: color
colorscheme delek




// create a new file: save vim setting
vim ~/.vimrc

4k



// view mode: undo 4 line
4u



// view mode: paste 4 times
4p




// view mode: jump to next word
w


// view mode: jump to back word
b



// view mode: jump to next letter
space


// view mode: jump to next line
j



// view mode: jump to up line
k

dw
